2.. class:: ShortestDistance 

3 :platform: Linux, MacOS, Windows 

4 :synopsis: The number of contacts between two atom groups 

5 

6.. classauthor:: Charlles Abreu <craabreu@gmail.com> 

7 

8""" 

9 

10import typing as t 

11 

12import openmm 

13from openmm import unit as mmunit 

14 

15from .collective_variable import CollectiveVariable 

16from .units import Quantity

19class ShortestDistance(CollectiveVariable, openmm.CustomCVForce): 

20 r""" 

21 A smooth approximation of the shortest distance between two atom groups: 

22 

23 .. math:: 

24 r_{\rm min}({\bf r}) = \frac{ 

25 \sum_{i \in {\bf g}_1} \sum_{j \in {\bf g}_2} 

26 r_{ij} e^{-\frac{r_{ij}^2}{2 \sigma^2}} 

27 }{ 

28 \sum_{i \in {\bf g}_1} \sum_{j \in {\bf g}_2} 

29 e^{-\frac{r_{ij}^2}{2 \sigma^2}} 

30 } 

31 

32 where :math:`r_{ij} = \|{\bf r}_j - {\bf r}_i\|` is the distance between atoms 

33 :math:`i` and :math:`j` and :math:`\sigma` is a parameter that controls the 

34 degree of approximation. The smaller the value of :math:`\sigma`, the closer the 

35 approximation to the true shortest distance. 

36 

37 In practice, a cutoff distance :math:`r_c` is applied to the atom pairs so that 

38 the collective variable is computed only for pairs of atoms separated by a distance 

39 less than :math:`r_c`. A switching function is also applied to smoothly turn off 

40 the collective variable starting from a distance :math:`r_s < r_c`. 

41 

42 .. note:: 

43 

44 Atoms are allowed to be in both groups. In this case, terms for which 

45 :math:`i = j` are ignored. 

46 

47 Parameters 

48 ---------- 

49 group1 

50 The indices of the atoms in the first group. 

51 group2 

52 The indices of the atoms in the second group. 

53 numAtoms 

54 The total number of atoms in the system, including those that are not in any 

55 of the groups. 

56 sigma 

57 The parameter that controls the degree of approximation. 

58 magnitude 

59 The expected order of magnitude of the shortest distance. This parameter does 

60 not affect the value of the collective variable, but a good estimate can 

61 improve numerical stability. 

62 cutoffDistance 

63 The cutoff distance for evaluating atom pairs. 

64 switchDistance 

65 The distance at which the switching function starts to be applied. 

66 pbc 

67 Whether to consider periodic boundary conditions in distance calculations. 

68 name 

69 The name of the collective variable. 

70 

71 Example 

72 ------- 

73 >>> import cvpack 

74 >>> import numpy as np 

75 >>> from openmm import unit 

76 >>> from openmmtools import testsystems 

77 >>> model = testsystems.HostGuestVacuum() 

78 >>> group1, group2 = [], [] 

79 >>> for residue in model.topology.residues(): 

80 ... group = group1 if residue.name == "B2" else group2 

81 ... group.extend(atom.index for atom in residue.atoms()) 

82 >>> coords = model.positions.value_in_unit(mmunit.nanometers) 

83 >>> np.linalg.norm( 

84 ... coords[group1, None, :] - coords[None, group2, :], 

85 ... axis=-1, 

86 ... ).min() 

87 0.17573... 

88 >>> num_atoms = model.system.getNumParticles() 

89 >>> min_dist = cvpack.ShortestDistance(group1, group2, num_atoms) 

90 >>> min_dist.addToSystem(model.system) 

91 >>> platform = openmm.Platform.getPlatformByName("Reference") 

92 >>> integrator = openmm.VerletIntegrator(1.0 * mmunit.femtoseconds) 

93 >>> context = openmm.Context(model.system, integrator, platform) 

94 >>> context.setPositions(model.positions) 

95 >>> min_dist.getValue(context) 

96 0.17573... nm 

97 """ 

98 

99 def __init__( 

100 self, 

101 group1: t.Sequence[int], 

102 group2: t.Sequence[int], 

103 numAtoms: int, 

104 sigma: mmunit.Quantity = Quantity(0.01 * mmunit.nanometers), 

105 magnitude: mmunit.Quantity = Quantity(0.2 * mmunit.nanometers), 

106 cutoffDistance: mmunit.Quantity = Quantity(0.5 * mmunit.nanometers), 

107 switchDistance: mmunit.Quantity = Quantity(0.4 * mmunit.nanometers), 

108 pbc: bool = True, 

109 name: str = "shortest_distance", 

110 ) -> None: 

111 if mmunit.is_quantity(sigma): 

112 sigma = sigma.value_in_unit(mmunit.nanometers) 

113 if mmunit.is_quantity(magnitude): 

114 magnitude = magnitude.value_in_unit(mmunit.nanometers) 

115 weight = f"exp(-0.5*(r^2 - {magnitude**2})/{sigma**2})" 

116 forces = { 

117 "numerator": openmm.CustomNonbondedForce(f"r*{weight}"), 

118 "denominator": openmm.CustomNonbondedForce(weight), 

119 } 

120 super().__init__("numerator/denominator") 

121 for cv, force in forces.items(): 

122 force.setNonbondedMethod( 

123 force.CutoffPeriodic if pbc else force.CutoffNonPeriodic 

124 ) 

125 force.setCutoffDistance(cutoffDistance) 

126 force.setUseSwitchingFunction(True) 

127 force.setSwitchingDistance(switchDistance) 

128 force.setUseLongRangeCorrection(False) 

129 for _ in range(numAtoms): 

130 force.addParticle([]) 

131 force.addInteractionGroup(group1, group2) 

132 self.addCollectiveVariable(cv, force) 

133 self._registerCV( 

134 name, 

135 mmunit.nanometers, 

136 group1=group1, 

137 group2=group2, 

138 numAtoms=numAtoms, 

139 sigma=sigma, 

140 magnitude=magnitude, 

141 cutoffDistance=cutoffDistance, 

142 switchDistance=switchDistance, 

143 pbc=pbc, 

144 ) 

145 

146 

147ShortestDistance.registerTag("!cvpack.ShortestDistance")
