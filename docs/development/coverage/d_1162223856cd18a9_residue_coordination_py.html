20class ResidueCoordination(openmm.CustomCentroidBondForce, BaseCollectiveVariable): 

21 r""" 

22 The number of contacts between two disjoint groups of residues: 

23 

24 .. math:: 

25 N({\bf r}) = \sum_{i \in {\bf G}_1} \sum_{j \in {\bf G}_2} S\left( 

26 \frac{\|{\bf R}_j({\bf r}) - {\bf R}_i({\bf r})\|}{r_0} 

27 \right) 

28 

29 where :math:`{\bf G}_1` and :math:`{\bf G}_2` are the two groups of residues, 

30 :math:`{\bf R}_i` is the centroid of the residue :math:`i`, :math:`r_0` is the 

31 threshold distance for defining a contact and :math:`S(x)` is a step function equal 

32 to :math:`1` if a contact is made or equal to :math:`0` otherwise. For trajectory 

33 analysis, it is fine to make :math:`S(x) = H(1-x)`, where `H` is the `Heaviside step 

34 function <https://en.wikipedia.org/wiki/Heaviside_step_function>`_. For molecular 

35 dynamics, however, :math:`S(x)` should be a continuous approximation of 

36 :math:`H(1-x)` for :math:`x \geq 0`. By default :cite:`Iannuzzi_2003`, the 

37 following function is used: 

38 

39 .. math:: 

40 

41 S(x) = \frac{1-x^6}{1-x^{12}} = \frac{1}{1+x^6} 

42 

43 Parameters 

44 ---------- 

45 residueGroup1 

46 The residues in the first group 

47 residueGroup2 

48 The residues in the second group 

49 pbc 

50 Whether the system has periodic boundary conditions 

51 stepFunction 

52 The function "step(1-x)" (for analysis only) or a continuous approximation 

53 thereof 

54 thresholdDistance 

55 The threshold distance (:math:`r_0`) for considering two residues as being 

56 in contact 

57 normalize 

58 Whether the number of contacts should be normalized by the total number of 

59 possible contacts 

60 weighByMass 

61 Whether the centroid of each residue should be weighted by the mass of the 

62 atoms in the residue 

63 includeHydrogens 

64 Whether hydrogen atoms should be included in the centroid calculations 

65 

66 Raises 

67 ------ 

68 ValueError 

69 If the two groups of residues are not disjoint 

70 

71 Example 

72 ------- 

73 >>> import cvpack 

74 >>> import itertools as it 

75 >>> import openmm 

76 >>> from openmm import app, unit 

77 >>> from openmmtools import testsystems 

78 >>> model = testsystems.LysozymeImplicit() 

79 >>> group1 = list(it.islice(model.topology.residues(), 125, 142)) 

80 >>> print(*[r.name for r in group1]) # doctest: +ELLIPSIS 

81 TRP ASP GLU ... ASN GLN THR 

82 >>> group2 = list(it.islice(model.topology.residues(), 142, 156)) 

83 >>> print(*[r.name for r in group2]) # doctest: +ELLIPSIS 

84 PRO ASN ARG ... ARG THR GLY 

85 >>> residue_coordination = cvpack.ResidueCoordination( 

86 ... group1, 

87 ... group2, 

88 ... stepFunction="step(1-x)", 

89 ... thresholdDistance=0.6*unit.nanometers, 

90 ... ) 

91 >>> residue_coordination.setUnusedForceGroup(0, model.system) 

92 1 

93 >>> model.system.addForce(residue_coordination) 

94 6 

95 >>> platform = openmm.Platform.getPlatformByName('Reference') 

96 >>> context = openmm.Context( 

97 ... model.system, openmm.CustomIntegrator(0), platform 

98 ... ) 

99 >>> context.setPositions(model.positions) 

100 >>> print(residue_coordination.getValue(context)) 

101 26.0 dimensionless 

102 >>> residue_coordination.setReferenceValue(26 * unit.dimensionless) 

103 >>> context.reinitialize(preserveState=True) 

104 >>> print(residue_coordination.getValue(context, digits=6)) 

105 1.0 dimensionless 

106 """

110 @mmunit.convert_quantities 

111 def __init__( # pylint: disable=too-many-arguments 

112 self, 

113 residueGroup1: t.Iterable[Residue], 

114 residueGroup2: t.Iterable[Residue], 

115 pbc: bool = True, 

116 stepFunction: str = "1/(1+x^6)", 

117 thresholdDistance: mmunit.ScalarQuantity = mmunit.Quantity( 

118 1.0, mmunit.nanometers 

119 ), 

120 normalize: bool = False, 

121 weighByMass: bool = True, 

122 includeHydrogens: bool = True, 

123 ) -> None: 

124 residueGroup1 = list(map(SerializableResidue, residueGroup1)) 

125 residueGroup2 = list(map(SerializableResidue, residueGroup2)) 

126 nr1 = len(residueGroup1) 

127 nr2 = len(residueGroup2) 

128 self._ref_val = nr1 * nr2 if normalize else 1.0 

129 expression = ( 

130 f"({stepFunction})/refval" 

131 f"; x=distance(g1,g2)/{thresholdDistance}" 

132 f"; refval={self._ref_val}" 

133 ) 

134 super().__init__(2, expression) 

135 self.setUsesPeriodicBoundaryConditions(pbc) 

136 for res in residueGroup1 + residueGroup2: 

137 atoms = [ 

138 atom.index 

139 for atom in res.atoms() 

140 if includeHydrogens or atom.element != "H" 

141 ] 

142 self.addGroup( 

143 atoms, 

144 *([] if weighByMass else [[1] * len(atoms)]), 

145 ) 

146 for idx1 in range(nr1): 

147 for idx2 in range(nr1, nr1 + nr2): 

148 self.addBond([idx1, idx2], []) 

149 self._registerCV( 

150 mmunit.dimensionless, 

151 residueGroup1, 

152 residueGroup2, 

153 pbc, 

154 stepFunction, 

155 thresholdDistance, 

156 normalize, 

157 weighByMass, 

158 includeHydrogens, 

159 ) 

160 

161 def getReferenceValue(self) -> mmunit.Quantity: 

162 """ 

163 Get the reference value used for normalizing the residue coordination. 

164 

165 Returns 

166 ------- 

167 mmunit.Quantity 

168 The reference value. 

169 """ 

170 return self._ref_val * self.getUnit()

186 self._ref_val = value
